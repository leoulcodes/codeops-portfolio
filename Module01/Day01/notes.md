#5 basic terminal commands 
    pwd - present working directory
    ls  - list files 
    touch file - create a file
    mkdir file - create a folder
    cp a b - copy file a into a file named b
#5 basic git commands
    git init - initialize a git repository
    git status - check the status of git tracking
    git add . - take the file into staging
    git commit -m "message" - save changes
    git push - push local changes.
